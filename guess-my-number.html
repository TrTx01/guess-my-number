<h3 id="result"></h3>

<script>

  const cards = [

    { value: 1, nums: [...Array(63)].map((_,i)=>i+1).filter(n=>n&1) },

    { value: 2, nums: [...Array(63)].map((_,i)=>i+1).filter(n=>n&2) },

    { value: 4, nums: [...Array(63)].map((_,i)=>i+1).filter(n=>n&4) },

    { value: 8, nums: [...Array(63)].map((_,i)=>i+1).filter(n=>n&8) },

    { value: 16, nums: [...Array(63)].map((_,i)=>i+1).filter(n=>n&16) },

    { value: 32, nums: [...Array(63)].map((_,i)=>i+1).filter(n=>n&32) }

  ];

  let index = 0;

  let total = 0;

  function showCard() {

    if (index < cards.length) {

      document.getElementById("numbers").innerText =

        cards[index].nums.join(", ");

    } else {

      document.getElementById("numbers").innerText = "";

      document.getElementById("result").innerText =

        "Your number is: " + total + " 🎉";

    }

  }

  function answer(yes) {

    if (yes) total += cards[index].value;

    index++;

    showCard();

  }

  showCard();

</script>
